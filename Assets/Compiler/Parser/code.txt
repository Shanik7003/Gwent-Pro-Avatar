effect 
{
    Name: "Damage",
    Params: {
        Amount: Number
    },
    Action: (targets, context) => 
	{
        for target in targets 
		{
            i = 0;
            while (i < Amount)
            {
                target.Power -= 1;
                i++;
            }

        };

    }
}
effect 
{
    Name: "Return" @ "ToDeck",
    Action: (targets, context) => 
    {
        for target in targets 
        {
            owner = target.Owner;
            deck = context.DeckOfPlayer(owner);
            deck.Push(target);
            deck.Shuffle();
            context.Board.Remove(target);
        };
    }
}
effect 
{
    Name: "Draw",
    Action: (targets, context) => 
    {
        topCard = context.Deck.Pop();
        context.Hand.Add(topCard);
        context.Hand.Shuffle();
    }
}
